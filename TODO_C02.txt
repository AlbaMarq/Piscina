#include <stdio.h>

void	lower(char str1, char str2)
{
	if (str2 < '1' || str2 > 'z')
		str1 = str1 - 32;
	else if (str2 > '9' && str2 < 'A')
		str1 = str1 - 32;
	else if (str2 > 'Z' && str2 < 'a')
		str1 = str1 - 32;
}

char	*ft_strcapitalize(char *str)
{
	int	i;

	i = 0;
	while (str[i] != '\0')
	{
		if (i == 0 && (str[0] >= 'a' && str[0] <= 'z'))
			str[0] = str[0] - 32;
		else if (str[i] >= 'A' && str[i] <= 'Z')
		{
			if (str[i - 1] >= 'a' && str[i - 1] <= 'z')
				str[i] = str[i] + 32;
			else if (str[i - 1] >= '1' || str[i - 1] <= '9')
				str[i] = str[i] + 32;
		}
		else if (str[i] >= 'a' && str[i] <= 'z')
		{
			lower(str[i], str[i - 1]);
		}
		i++;
	}
	return (str);
}

#include <stdio.h>

int	ft_strlen(char *str)
{
	int	len;

	len = 0;
	while (*str != '\0')
	{
		len++;
		str++;
	}
	return (len);
}

unsigned int ft_strlcpy(char *dest, char *src, unsigned int size)
{
    unsigned int	i;
    unsigned int    dest_len;
    unsigned int    src_len;

	i = 0;
	while (src[i] != '\0' && i < size - 1)
	{
		dest[i] = src[i];
		i++;
	}
    dest_len = ft_strlen(dest);
    if (dest_len > 0)
    {
        dest[i] = '\0';
    }
    src_len = ft_strlen(src);
	return (src_len);
}

#include <stdio.h>
#include <unistd.h>

void	ft_putstr_non_printable(char *str)
{
	int		i;
	char	div_hex;
	char	mod_hex;
	char	*hex;

	i = 0;
	div_hex = 0;
	mod_hex = 0;
	hex = "0123456789abcdef";
	while (*str != '\0')
	{
		if (*str < 32 || *str > 126)
		{
			div_hex = hex[*str / 16];
			mod_hex = hex[*str % 16];
			write(1, "\\", 1);
			write(1, &div_hex, 1);
			write(1, &mod_hex, 1);
		}
		else
			write(1, &str[i], 1);
		str++;
	}
}
