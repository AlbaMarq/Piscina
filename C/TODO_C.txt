#include <unistd.h>

void    concat(char a, char b, char c, char d)
{
    char    coma;
    char    esp;

    coma = ',';
    esp = ' ';
    write(1, &a, 1);
    write(1, &b, 1);
    write(1, &esp, 1);
    write(1, &c, 1);
    write(1, &d, 1);
    if (a != '9' || b != '8' || c != '9' || d != '9')
    {
        write(1, &coma, 1);
        write(1, &esp, 1);
    }
}

void    buclewhile(char a, char b, char c, char d)
    write(1, &esp, 1);
    write(1, &c, 1);
    write(1, &d, 1);
    if (a != '9' || b != '8' || c != '9' || d != '9')
    {
        write(1, &coma, 1);
        write(1, &esp, 1);
    }
}

void    buclewhile(char a, char b, char c, char d)
{
    while (a <= '9')
    {
        while (b <= '8')
        {
            while (c <= '9')
            {
                while (d <= '9')
                {
                    concat(a, b, c, d);
                    d++;
                }
                d = '0';
                c++;
            }
            c = '0';
            b++;
        }
        b = '0';
        a++;
    }
}

    a = '0';
